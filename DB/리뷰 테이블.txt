--- 리뷰 테이블
CREATE TABLE UserReview (
    review_id    NUMBER PRIMARY KEY,
    user_id      VARCHAR2(60) NOT NULL,
    user_nickname VARCHAR2(20),
    book_id      NUMBER NOT NULL,
    review_rating NUMBER(1) CHECK (review_rating BETWEEN 1 AND 5),
    review_content VARCHAR2(1000),
    review_date  DATE DEFAULT SYSDATE,
    CONSTRAINT fk_review_user FOREIGN KEY (user_id)
        REFERENCES users (user_id),
    CONSTRAINT fk_review_book FOREIGN KEY (book_id)
        REFERENCES Book (book_id)
);

-- UserReview 테이블용 시퀀스 생성
CREATE SEQUENCE USERREVIEW_SEQ
  START WITH 1
  INCREMENT BY 1
  NOCACHE
  NOCYCLE;
